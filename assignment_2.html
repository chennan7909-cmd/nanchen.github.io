import numpy as np
import plotly.graph_objects as go
# If running outside Jupyter and nothing appears, uncomment:
# import plotly.io as pio; pio.renderers.default = "browser"

# ---------------------------------------------------------
# Nightingale 3D Interactive Redesign (no HTML file)
# ---------------------------------------------------------
# - Three stacked categories (Disease / Wounds / Other)
# - Radius ~ sqrt(value) so wedge area ~ value (like original)
# - fig.show() will open the interactive 3D plot directly
# ---------------------------------------------------------

# 1) Labels (12 months)
months = [
    "Apr 1854","May 1854","Jun 1854","Jul 1854","Aug 1854","Sep 1854",
    "Oct 1854","Nov 1854","Dec 1854","Jan 1855","Feb 1855","Mar 1855"
]

# 2) Sample data (replace with historical series if desired)
disease = np.array([1100, 950, 840, 780, 720, 680, 640, 600, 560, 520, 480, 440], dtype=float)
wounds  = np.array([ 120, 150, 130, 140, 160, 120, 110, 100,  90,  85,  80,  75], dtype=float)
other   = np.array([  80,  90,  85,  82,  78,  74,  70,  66,  62,  60,  58,  55], dtype=float)

layers = [
    ("Disease", disease, "#5DADE2"),  # blue
    ("Wounds",  wounds,  "#F5B7B1"),  # pink
    ("Other",   other,   "#95A5A6"),  # gray
]

# 3) Helper: build a 3D polar wedge surface at height z
def wedge_surface(theta0, theta1, r0, r1, z, n_theta=36, n_r=8):
    theta = np.linspace(theta0, theta1, n_theta)
    r = np.linspace(r0, r1, n_r)
    T, R = np.meshgrid(theta, r, indexing="xy")
    X = R * np.cos(T)
    Y = R * np.sin(T)
    Z = np.full_like(X, z, dtype=float)
    return X, Y, Z

# 4) Geometry/scales
scale = 0.035                 # radius  sqrt(value)  area  value
dtheta = 2 * np.pi / len(months)
gap = 0.04                    # angular gap between wedges
z_gap = 0.18                  # vertical spacing between layers
layer_z = {"Disease": 0.0, "Wounds": z_gap, "Other": 2*z_gap}

# 5) Build figure
fig = go.Figure()

for mi, month in enumerate(months):
    theta0 = mi * dtheta + gap/2
    theta1 = (mi + 1) * dtheta - gap/2

    cumulative = 0.0
    for name, series, color in layers:
        value = float(series[mi])
        r_in  = scale * np.sqrt(max(cumulative, 0.0))
        r_out = scale * np.sqrt(max(cumulative + value, 0.0))
        X, Y, Z = wedge_surface(theta0, theta1, r_in, r_out, layer_z[name])

        fig.add_trace(go.Surface(
            x=X, y=Y, z=Z,
            showscale=False,
            opacity=0.95 if name=="Disease" else (0.92 if name=="Wounds" else 0.90),
            surfacecolor=np.ones_like(Z),
            colorscale=[[0, color], [1, color]],
            hovertemplate=f"<b>{month}</b><br>Layer: {name}<br>{name}: {value:.0f}<extra></extra>",
            name=name,
            legendgroup=name,
        ))
        cumulative += value

# 6) Layer-toggle buttons
buttons = []
for label in ["All", "Disease", "Wounds", "Other"]:
    visible = []
    for tr in fig.data:
        visible.append(True if label == "All" else tr.name == label)
    buttons.append(dict(label=label, method="update", args=[{"visible": visible}]))

# 7) Month highlight slider (dims non-selected months)
steps = []
surfaces_per_month = len(layers)
for mi, month in enumerate(months):
    opac = []
    for idx, _ in enumerate(fig.data):
        opac.append(0.98 if idx // surfaces_per_month == mi else 0.50)
    steps.append(dict(
        label=month.split()[0],
        method="update",
        args=[{"opacity": opac}]
    ))

# 8) Layout and camera
fig.update_layout(
    title="Diagram of the Causes of Mortality  3D Interactive Redesign",
    scene=dict(
        xaxis=dict(visible=False),
        yaxis=dict(visible=False),
        zaxis=dict(title="Layer Height", gridcolor="lightgray", zerolinecolor="lightgray"),
        aspectmode="manual",
        aspectratio=dict(x=1, y=1, z=0.5),
    ),
    margin=dict(l=0, r=0, t=60, b=0),
    updatemenus=[dict(type="buttons", direction="right", x=0.5, y=1.12,
                      buttons=buttons, xanchor="center", yanchor="top")],
    sliders=[dict(active=0, currentvalue=dict(prefix="Highlight: "), steps=steps, y=1.02)],
    showlegend=False
)

fig.update_scenes(camera=dict(eye=dict(x=1.6, y=1.6, z=1.0)))

# 9) Show the interactive 3D plot directly (no HTML file saved)
fig.show()
